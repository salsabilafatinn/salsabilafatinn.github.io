import tkinter as tk
import matplotlib.pyplot as plt
from matplotlib.backends.backend_tkagg import FigureCanvasTkAgg

df = pd.read_csv("Advertising.csv")

X = df[['TV','radio','newspaper']]
Y = df['sales']

regr = linear_model.LinearRegression()
regr.fit(X, Y)

print('Intercept: \n', regr.intercept_)
print('Coefficients: \n', regr.coef_)

# tkinter GUI
root= tk.Tk()
root.config(background="#F3CCFF")
canvas1 = tk.Canvas(root, background="#F3CCFF")
canvas1.pack()

Intercept_result = ('Intercept: ', regr.intercept_)
label_Intercept = tk.Label(root, text=Intercept_result, bg="#FFFFD0",font=('times',11))
canvas1.create_window(190, 170, window=label_Intercept)

Coefficients_result  = ('Coefficients: ', regr.coef_)
label_Coefficients = tk.Label(root, text=Coefficients_result, bg="#FFFFD0", font=('times',11))
canvas1.create_window(190, 200, window=label_Coefficients)

# New_TV label dan input box
label1 = tk.Label(root, text='TV: ', bg="#FFFFD0", font=('times',11))
canvas1.create_window(100, 35, window=label1)

entry1 = tk.Entry (root) #membuat entry box 1
canvas1.create_window(250, 35, window=entry1)

# New_radio Label dan input box
label2 = tk.Label(root, text='Radio: ', bg="#FFFFD0", font=('times',11))
canvas1.create_window(100, 65, window=label2)

entry2 = tk.Entry (root) #membuat entry box 1
canvas1.create_window(250, 65, window=entry2)

# New_newspaper Label and input box
label3 =tk.Label(root, text='Newspaper: ', bg="#FFFFD0", font=('times',11))
canvas1.create_window(100, 95, window=label3)

entry3 = tk.Entry (root) #membuat entry box 1
canvas1.create_window(250, 95, window=entry3)


def values(): 
    global TV#our 1st input variable
    TV = float(entry1.get())
    
    global radio #our 2nd input variable
    radio = float(entry2.get())
    
    global newspaper
    newspaper = float(entry3.get())
    
    Prediction_result  = ('Predicted Sales: ', regr.predict([[TV, radio, newspaper]]))
    label_Prediction = tk.Label(root, text= Prediction_result, bg="#FFADBC", font=('times',12,'bold'))
    canvas1.create_window(180, 240, window=label_Prediction)
    
button1 = tk.Button (root, text='Predict Sales',command=values, bg="#FFADBC", font=('times',12,'bold'))
canvas1.create_window(200, 130, window=button1)
 
                          
#scatter plot 1
figure3 = plt.Figure(figsize=(5,4), dpi=100)
ax3 = figure3.add_subplot(111)
ax3.scatter(df['TV'].astype(float), df['sales'].astype(float), color= 'blue')
scatter3 = FigureCanvasTkAgg(figure3, root) 
scatter3.get_tk_widget().pack(side=tk.RIGHT, fill=tk.BOTH)
ax3.legend(['sales'])
ax3.set_xlabel('TV')
ax3.set_title('Sales VS TV')

#scatter plot 2 
figure4 = plt.Figure(figsize=(5,4), dpi=100)
ax4 = figure4.add_subplot(111)
ax4.scatter(df['radio'].astype(float), df['sales'].astype(float), color= 'red')
scatter4 = FigureCanvasTkAgg(figure4, root) 
scatter4.get_tk_widget().pack(side=tk.RIGHT, fill=tk.BOTH)
ax4.legend(['sales'])
ax4.set_xlabel('radio')
ax4.set_title('Sales VS Radio')
                          
#scatter plot 3
figure4 = plt.Figure(figsize=(5,4), dpi=100)
